#include <SoftwareSerial.h>

SoftwareSerial BTSerial(10, 11); // RX | TX
const int bouton = 7;
const int led_rouge = 13;

byte var=0;


void setup() {


 pinMode(9, OUTPUT);
 digitalWrite(9, HIGH);

 Serial.begin(9600);
 Serial.println("Enter AT commands:");

 BTSerial.begin(38400); 

pinMode(bouton, INPUT);
pinMode(led_rouge, OUTPUT);

}

void loop() {

if(digitalRead(bouton)==HIGH) {

var ++;

switch(var) {

case 1:
connexion();
break;

case 2:
Serial.println("A ON"); 
BTSerial.write("1");
break;

case 3:
Serial.println("A OFF"); 
BTSerial.write("0");
var=0;
break;
}

while(digitalRead(bouton)==HIGH); // on bloque tant que le bouton reste appuyé
delay(50); // élimine les effets de rebond
}

gestionSerie();

}

void gestionSerie() {

 if (BTSerial.available())
 Serial.write(BTSerial.read());

 
 if (Serial.available())
 BTSerial.write(Serial.read());

}

void connexion() {

 Serial.println("Enter AT :");
 digitalWrite(led_rouge, HIGH); 
 BTSerial.print("AT\r\n");
 delay(200);
 BTSerial.print("AT+CMODE=0\r\n");
 delay(200);
 BTSerial.print("AT+ROLE=1\r\n");
 delay(200);
 BTSerial.print("AT+INIT\r\n");
 delay(200);
 BTSerial.print("AT + LINK=2014,6,301687\r\n");
 delay(200);
 BTSerial.print("AT+PSWD=1234\r\n");
 delay(5000);
 Serial.println("PRET"); 
 delay(5000);

}
